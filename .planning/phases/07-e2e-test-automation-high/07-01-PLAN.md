**A. Fixture generation script** — `src/e2e/fixtures/generate-e2e-fixtures.mjs`

This script creates the files that cannot be committed (large sparse file) and commits the small stubs. Run once before the test suite.

```js
#!/usr/bin/env node
/**
 * Generate E2E test fixtures.
 * Run: node src/e2e/fixtures/generate-e2e-fixtures.mjs
 *
 * Creates in test-fixtures-e2e/:
 *   large-sparse.pdf   — 110 MB sparse file (triggers >100 MB guard)
 *   large-sparse.jpg   — 110 MB sparse file (image variant)
 *
 * Stubs (committed as zero-byte files, created here for documentation):
 *   corrupt.pdf        — 0 bytes (corrupt file error path)
 *   corrupt.jpg        — 0 bytes (corrupt image error path)
 *
 * Note: test-fixtures-e2e/large-sparse.* are in .gitignore.
 * Re-run this script on a fresh checkout before running test:e2e.
 */
import { writeFileSync, mkdirSync, openSync, ftruncateSync, closeSync } from 'fs';
import { join, dirname } from 'path';
import { fileURLToPath } from 'url';

const __dirname = dirname(fileURLToPath(import.meta.url));
const outDir = join(__dirname, '../../../test-fixtures-e2e');
mkdirSync(outDir, { recursive: true });

function createSparseFile(filePath, sizeBytes) {
  const fd = openSync(filePath, 'w');
  ftruncateSync(fd, sizeBytes);
  closeSync(fd);
  console.log(`Created sparse file: ${filePath} (${(sizeBytes / 1024 / 1024).toFixed(0)} MB)`);
}

const MB_110 = 110 * 1024 * 1024;
createSparseFile(join(outDir, 'large-sparse.pdf'), MB_110);
createSparseFile(join(outDir, 'large-sparse.jpg'), MB_110);

// Zero-byte corrupt stubs (already committed, recreated here for safety)
writeFileSync(join(outDir, 'corrupt.pdf'), Buffer.alloc(0));
writeFileSync(join(outDir, 'corrupt.jpg'), Buffer.alloc(0));

console.log('E2E fixtures ready in test-fixtures-e2e/');
```

Run the script immediately after creating it: `node src/e2e/fixtures/generate-e2e-fixtures.mjs`

**B. Driver helpers** — `src/e2e/helpers/driver.ts`

Shared utilities for all E2E tests:

```ts
import type { Browser } from 'webdriverio';
import { mkdirSync, rmSync, existsSync } from 'fs';
import { join } from 'path';

export const E2E_OUTPUT_DIR = join(process.cwd(), 'e2e-output');
export const FIXTURES_DIR = join(process.cwd(), 'test-fixtures-e2e');
export const REAL_FIXTURES_DIR = join(process.cwd(), 'test-fixtures');

/** Ensure e2e-output dir exists and return a unique output path for this test. */
export function prepareOutputPath(filename: string): string {
  mkdirSync(E2E_OUTPUT_DIR, { recursive: true });
  const outPath = join(E2E_OUTPUT_DIR, filename);
  // Clean up any leftover from a previous run
  if (existsSync(outPath)) rmSync(outPath);
  return outPath;
}

/** Wait for the app's step bar to show the target step index (0=Pick, 1=Configure, 2=Compare, 3=Save). */
export async function waitForStep(browser: Browser, stepIndex: number): Promise<void> {
  // Step bar items have data-step attribute or aria-current="step" on active item
  // The StepBar component renders items with data-testid="step-{index}" or similar.
  // Use a CSS selector targeting the active step indicator.
  await browser.waitUntil(
    async () => {
      const active = await browser.$(`[data-testid="step-bar-item"][data-active="true"]`);
      if (!active) return false;
      const idx = await active.getAttribute('data-step-index');
      return idx === String(stepIndex);
    },
    { timeout: 15000, interval: 200, timeoutMsg: `Timed out waiting for step ${stepIndex}` }
  );
}

/** Wait for the loading overlay to disappear (file is loaded). */
export async function waitForFileLoaded(browser: Browser): Promise<void> {
  // App sets isLoading=true for 600ms on file select; wait for spinner to clear
  await browser.waitUntil(
    async () => {
      const spinner = await browser.$('[data-testid="loading-spinner"]');
      return !(await spinner.isDisplayed().catch(() => false));
    },
    { timeout: 10000, interval: 100, timeoutMsg: 'Timed out waiting for file load' }
  );
}

/** Wait for processing to complete (the Generate Preview button becomes enabled again). */
export async function waitForProcessingComplete(browser: Browser): Promise<void> {
  await browser.waitUntil(
    async () => {
      // Processing is done when CompareStep is visible
      const compare = await browser.$('[data-testid="compare-step"]');
      return await compare.isDisplayed().catch(() => false);
    },
    { timeout: 90000, interval: 500, timeoutMsg: 'Timed out waiting for processing to complete' }
  );
}

/** Take a screenshot on failure. Called from wdio afterEach hook. */
export async function screenshotOnFailure(browser: Browser, testTitle: string): Promise<void> {
  const dir = join(process.cwd(), '.e2e-artifacts', 'screenshots');
  mkdirSync(dir, { recursive: true });
  const safe = testTitle.replace(/[^a-z0-9]/gi, '_').slice(0, 80);
  await browser.saveScreenshot(join(dir, `${safe}_${Date.now()}.png`));
}
```

**C. Dialog mock helper** — `src/e2e/helpers/dialogs.ts`

The Tauri `@tauri-apps/plugin-dialog` `save()` call opens a native OS dialog. In tests we must bypass it. The approach: execute a JS snippet in the WebView via `browser.execute()` that patches `window.__TAURI_INTERNALS__.transformCallback` / `window.__TAURI__.dialog.save` before the test triggers a save.

```ts
import type { Browser } from 'webdriverio';

/**
 * Patches window.__TAURI__.dialog.save (Tauri v2 API) to resolve immediately
 * with the provided filePath instead of showing the OS dialog.
 *
 * Call BEFORE the user action that triggers the save dialog.
 */
export async function mockSaveDialog(browser: Browser, outputPath: string): Promise<void> {
  await browser.execute((path: string) => {
    // Tauri v2 exposes dialog via window.__TAURI__.dialog
    const tauri = (window as unknown as { __TAURI__?: { dialog?: { save?: unknown } } }).__TAURI__;
    if (tauri?.dialog) {
      // Patch: replace save() with a function that immediately resolves to the output path
      (tauri.dialog as Record<string, unknown>).save = async () => path;
    }
  }, outputPath);
}

/**
 * Removes the mock so subsequent calls use the real OS dialog.
 * Not required in tests (each test gets a fresh WebView), but available for cleanup.
 */
export async function clearSaveDialogMock(browser: Browser): Promise<void> {
  await browser.execute(() => {
    // Re-importing the plugin would reset it; just delete the patched method.
    const tauri = (window as unknown as { __TAURI__?: { dialog?: Record<string, unknown> } }).__TAURI__;
    if (tauri?.dialog) {
      delete tauri.dialog.save;
    }
  });
}
```

Note on the mock approach: Tauri v2 exposes plugin APIs under `window.__TAURI__`. The exact object path is `window.__TAURI__.dialog.save`. If inspection during test run shows a different path (e.g. `window.__TAURI_INTERNALS__`), update accordingly. The verify step below confirms the mock pattern works.
