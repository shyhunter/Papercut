**NOTE on file injection:** Tauri apps use native file dialogs — WebDriverIO cannot programmatically click "Open" in the OS dialog. Instead, the tests inject the file path directly by calling a JS `customEvent` or by calling the Tauri `open_file` command directly via `window.__TAURI__`. The recommended pattern:

```ts
async function injectFile(browser: Browser, filePath: string): Promise<void> {
  // Tauri v2: invoke the open_file flow by dispatching a custom synthetic event
  // OR by calling the same internal handler that the file-picker button calls.
  // The cleanest approach: patch window.__TAURI__.dialog.open to return the path,
  // then click the Open File button so App.tsx handleFileSelected() runs normally.
  await browser.execute((path: string) => {
    const tauri = (window as unknown as { __TAURI__?: { dialog?: Record<string, unknown> } }).__TAURI__;
    if (tauri?.dialog) {
      tauri.dialog.open = async () => path;
    }
  }, filePath);
  const openBtn = await browser.$('[data-testid="open-file-btn"]');
  await openBtn.click();
}
```

import { browser } from '@wdio/globals';
import { existsSync, statSync } from 'fs';
import { join } from 'path';
import { mockSaveDialog } from '../helpers/dialogs';
import {
  waitForStep,
  waitForProcessingComplete,
  screenshotOnFailure,
  prepareOutputPath,
  FIXTURES_DIR,
  REAL_FIXTURES_DIR,
} from '../helpers/driver';

const PHOTO_PDF = join(REAL_FIXTURES_DIR, 'photo_heavy.pdf');
const TEXT_PDF = join(REAL_FIXTURES_DIR, 'warnock_camelot.pdf');
const LARGE_PDF = join(FIXTURES_DIR, 'large-sparse.pdf');
const CORRUPT_PDF = join(FIXTURES_DIR, 'corrupt.pdf');

async function injectFile(filePath: string): Promise<void> {
  await browser.execute((path: string) => {
    const tauri = (window as unknown as { __TAURI__?: { dialog?: Record<string, unknown> } }).__TAURI__;
    if (tauri?.dialog) { tauri.dialog.open = async () => path; }
  }, filePath);
  const openBtn = await browser.$('[data-testid="open-file-btn"]');
  await openBtn.click();
}

async function navigateToCompare(): Promise<void> {
  await waitForStep(browser, 1);
  const generateBtn = await browser.$('[data-testid="generate-preview-btn"]');
  await generateBtn.waitForClickable({ timeout: 5000 });
  await generateBtn.click();
  await waitForProcessingComplete(browser);
  await waitForStep(browser, 2);
}

afterEach(async function (this: Mocha.Context) {
  if (this.currentTest?.state === 'failed') {
    await screenshotOnFailure(browser, this.currentTest.fullTitle());
  }
  // Return to landing for next test
  try {
    const processAnother = await browser.$('[data-testid="process-another-btn"]');
    if (await processAnother.isDisplayed().catch(() => false)) {
      await processAnother.click();
    }
  } catch { /* already on landing */ }
});

// ─── PDF COMPRESSION QUALITY MATRIX ─────────────────────────────────────────

describe('PDF compression — quality levels', () => {
  for (const quality of ['web', 'screen', 'print', 'archive'] as const) {
    it(`[PDF-Q-${quality.toUpperCase()}] processes photo_heavy.pdf at quality=${quality}`, async () => {
      const outPath = prepareOutputPath(`pdf-quality-${quality}.pdf`);
      await injectFile(PHOTO_PDF);
      await waitForStep(browser, 1);

      // Select quality level
      const qualitySelect = await browser.$('[data-testid="quality-select"]');
      await qualitySelect.selectByAttribute('value', quality);

      await mockSaveDialog(browser, outPath);
      await navigateToCompare();

      // Verify Compare step rendered
      const compareStep = await browser.$('[data-testid="compare-step"]');
      await expect(compareStep).toBeDisplayed();

      // Click Save
      const saveBtn = await browser.$('[data-testid="save-btn"]');
      await saveBtn.click();
      await waitForStep(browser, 3);

      // Verify output file exists and is a valid PDF
      expect(existsSync(outPath), `Output PDF not found at ${outPath}`).toBe(true);
      const outSize = statSync(outPath).size;
      expect(outSize, 'Output PDF must be > 0 bytes').toBeGreaterThan(0);
      // Output should be smaller than input for photo_heavy.pdf (compression expected)
      const inSize = statSync(PHOTO_PDF).size;
      expect(outSize, `web/screen quality should compress photo PDF (out=${outSize}, in=${inSize})`).toBeLessThan(inSize);
    });
  }
});

// ─── PDF ERROR PATHS ─────────────────────────────────────────────────────────

describe('PDF error paths', () => {
  it('[PDF-ERR-OVERSIZE] >100 MB file shows blocking modal', async () => {
    await injectFile(LARGE_PDF);
    // Should NOT advance to Configure — stays on step 0
    await browser.pause(1500); // allow async size check to complete
    const modal = await browser.$('[data-testid="file-size-limit-modal"]');
    await expect(modal).toBeDisplayed();
    // Dismiss modal; app remains on landing
    const dismissBtn = await browser.$('[data-testid="file-size-limit-dismiss"]');
    await dismissBtn.click();
    await browser.pause(500);
    await expect(modal).not.toBeDisplayed();
  });

  it('[PDF-ERR-CORRUPT] zero-byte file shows inline error', async () => {
    await injectFile(CORRUPT_PDF);
    await browser.pause(1500);
    // Either empty-file-error or corrupt-file-error should appear; both are zero-byte paths
    const emptyErr = await browser.$('[data-testid="empty-file-error"]');
    const corruptErr = await browser.$('[data-testid="corrupt-file-error"]');
    const oneVisible =
      (await emptyErr.isDisplayed().catch(() => false)) ||
      (await corruptErr.isDisplayed().catch(() => false));
    expect(oneVisible, 'Expected an inline error for zero-byte file').toBe(true);
    // App must NOT have advanced to Configure step
    const configureStep = await browser.$('[data-testid="configure-step"]');
    expect(await configureStep.isDisplayed().catch(() => false)).toBe(false);
  });

  it('[PDF-ERR-CANCEL] cancel mid-processing returns to Configure step', async () => {
    await injectFile(PHOTO_PDF);
    await waitForStep(browser, 1);

    const generateBtn = await browser.$('[data-testid="generate-preview-btn"]');
    await generateBtn.click();
    // Click cancel immediately (within processing window)
    await browser.pause(300);
    const cancelBtn = await browser.$('[data-testid="cancel-btn"]');
    if (await cancelBtn.isDisplayed().catch(() => false)) {
      await cancelBtn.click();
    }
    // After cancel, app may go to Compare (showing cancelled state) or stay at Configure
    // Either way it must NOT crash; wait briefly
    await browser.pause(2000);
    const configureVisible = await browser.$('[data-testid="configure-step"]').isDisplayed().catch(() => false);
    const compareVisible = await browser.$('[data-testid="compare-step"]').isDisplayed().catch(() => false);
    expect(
      configureVisible || compareVisible,
      'After cancel, app must show either Configure or Compare step (not blank)'
    ).toBe(true);
  });
});

// ─── PDF SAVE DIALOG FILTER ───────────────────────────────────────────────────

describe('PDF save dialog filter', () => {
  it('[PDF-SAVE-FILTER] save dialog uses PDF filter, not image filter', async () => {
    const outPath = prepareOutputPath('pdf-save-filter-test.pdf');
    await injectFile(PHOTO_PDF);
    await waitForStep(browser, 1);

    // Intercept the dialog.save call and capture the filters it was called with
    let capturedFilters: unknown = null;
    await browser.execute(() => {
      const tauri = (window as unknown as { __TAURI__?: { dialog?: Record<string, unknown> } }).__TAURI__;
      if (tauri?.dialog) {
        const orig = tauri.dialog.save as Function;
        tauri.dialog.save = async (opts: unknown) => {
          (window as unknown as { __E2E_SAVE_OPTS__?: unknown }).__E2E_SAVE_OPTS__ = opts;
          return null; // cancel — we only want to inspect the filters
        };
      }
    });

    await navigateToCompare();
    const saveBtn = await browser.$('[data-testid="save-btn"]');
    await saveBtn.click();
    await browser.pause(500);

    capturedFilters = await browser.execute(() =>
      (window as unknown as { __E2E_SAVE_OPTS__?: unknown }).__E2E_SAVE_OPTS__
    );

    // The save options must contain a filter with 'pdf' extension (case-insensitive)
    const filters = JSON.stringify(capturedFilters).toLowerCase();
    expect(filters, 'Save dialog filters must mention "pdf"').toContain('pdf');
    expect(filters, 'Save dialog filters must NOT mention "jpeg" or "png"').not.toMatch(/jpeg|png|webp/);
  });
});
