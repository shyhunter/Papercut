<objective>
Add React error boundaries at the step level and app level so unexpected render failures show a recoverable fallback UI instead of a blank screen.

Purpose: Without error boundaries, any runtime render exception in ConfigureStep, CompareStep, or SaveStep produces a completely blank screen with no recovery path. This plan adds two boundary types: StepErrorBoundary (wraps individual steps, shows Reset this step) and AppErrorBoundary (wraps the whole app, shows Restart app).
Output: ErrorBoundary.tsx with both boundary classes, App.tsx updated to wrap all steps, tests covering throw → fallback → reset cycle.
</objective>

  <behavior>
    StepErrorBoundary:
    - Class component extending React.Component (required — hooks cannot catch render errors)
    - Props: `children: React.ReactNode`, `stepName: string` (e.g. "Configure", "Compare", "Save")
    - State: `{ hasError: boolean; error: Error | null; showDetails: boolean }`
    - Fallback UI when hasError=true:
      ```
      [rounded card, border-destructive/30 bg-destructive/5, p-6, flex col, gap-4]
        [header row]
          [AlertCircle icon text-destructive h-5 w-5]
          <p class="text-sm font-medium text-foreground">{stepName} encountered an unexpected error.</p>
        [collapsed details section — togglable]
          <button type="button" onClick toggle>Show error details / Hide error details</button>
          {showDetails && <pre class="text-xs text-muted-foreground overflow-auto max-h-40 p-2 rounded bg-muted">{error.message}</pre>}
        [Reset this step button — outline, sm]
          onClick: setState({ hasError: false, error: null, showDetails: false })
      ```
    - The Reset this step button clears the boundary state so the step re-renders. App state (fileEntry, etc.) is NOT touched. If the same error re-occurs, the boundary catches it again.

    AppErrorBoundary:
    - Same pattern as StepErrorBoundary but:
      - Full-width (no max-width constraint)
      - Fallback message: "The app encountered an unexpected error."
      - Recovery button: "Restart app" — calls `window.location.reload()` (since App-level reset is not reliable without re-mounting the full React tree)
      - Same collapsible error details section

    Wrapping in App.tsx:
    - Wrap the entire return of App() in `<AppErrorBoundary>...</AppErrorBoundary>`.
    - Wrap each step block individually with StepErrorBoundary:
      - `<StepErrorBoundary stepName="Configure">` around both ConfigureStep and ImageConfigureStep blocks
      - `<StepErrorBoundary stepName="Compare">` around both CompareStep and ImageCompareStep blocks
      - `<StepErrorBoundary stepName="Save">` around both SaveStep blocks
    - StepBar, LandingCard, PrivacyFooter, and Toaster remain outside step boundaries (they are simple and stable).

    Error details format (Claude's discretion): show `error.message` only (not full stack trace) — friendly and precise without overwhelming. A stack trace is available for debug but not shown by default.

    Cases → expected behavior:
    - ConfigureStep throws during render → StepErrorBoundary fallback with "Configure encountered an unexpected error." + Reset this step
    - CompareStep throws during render → StepErrorBoundary fallback with "Compare encountered an unexpected error." + Reset this step
    - Root App throws → AppErrorBoundary fallback with full-width "The app encountered an unexpected error." + Restart app
    - Reset this step → boundary clears, step re-renders (if the error was transient, it works; if persistent, it catches again)

    ```tsx
    import React from 'react';
    import { AlertCircle } from 'lucide-react';
    import { Button } from '@/components/ui/button';
    import { cn } from '@/lib/utils';

    // ── Shared fallback UI ─────────────────────────────────────────────────────

    interface BoundaryFallbackProps {
      message: string;
      error: Error | null;
      showDetails: boolean;
      onToggleDetails: () => void;
      recoveryLabel: string;
      onRecover: () => void;
      fullWidth?: boolean;
    }

    function BoundaryFallback({
      message, error, showDetails, onToggleDetails, recoveryLabel, onRecover, fullWidth,
    }: BoundaryFallbackProps) {
      return (
        <div className={cn(
          'flex flex-1 items-center justify-center p-8',
          fullWidth && 'w-full',
        )}>
          <div className={cn(
            'rounded-lg border border-destructive/30 bg-destructive/5 p-6 flex flex-col gap-4',
            fullWidth ? 'w-full' : 'max-w-md w-full',
          )}>
            <div className="flex items-center gap-2">
              <AlertCircle className="h-5 w-5 text-destructive flex-none" />
              <p className="text-sm font-medium text-foreground">{message}</p>
            </div>
            <div className="flex flex-col gap-2">
              <button
                type="button"
                onClick={onToggleDetails}
                className="text-xs text-muted-foreground hover:text-foreground text-left transition-colors"
              >
                {showDetails ? 'Hide error details' : 'Show error details'}
              </button>
              {showDetails && error && (
                <pre className="text-xs text-muted-foreground overflow-auto max-h-40 p-2 rounded bg-muted whitespace-pre-wrap break-words">
                  {error.message}
                </pre>
              )}
            </div>
            <Button variant="outline" size="sm" onClick={onRecover} className="self-start">
              {recoveryLabel}
            </Button>
          </div>
        </div>
      );
    }

    // ── StepErrorBoundary ──────────────────────────────────────────────────────

    interface StepBoundaryProps {
      children: React.ReactNode;
      stepName: string;
    }

    interface StepBoundaryState {
      hasError: boolean;
      error: Error | null;
      showDetails: boolean;
    }

    export class StepErrorBoundary extends React.Component<StepBoundaryProps, StepBoundaryState> {
      constructor(props: StepBoundaryProps) {
        super(props);
        this.state = { hasError: false, error: null, showDetails: false };
      }

      static getDerivedStateFromError(error: Error): StepBoundaryState {
        return { hasError: true, error, showDetails: false };
      }

      componentDidCatch(error: Error, info: React.ErrorInfo) {
        // Error logged to boundary state — no external error service (privacy-first)
        console.error('[ErrorBoundary]', this.props.stepName, error, info);
      }

      handleReset = () => {
        this.setState({ hasError: false, error: null, showDetails: false });
      };

      handleToggleDetails = () => {
        this.setState((prev) => ({ showDetails: !prev.showDetails }));
      };

      render() {
        if (this.state.hasError) {
          return (
            <BoundaryFallback
              message={`${this.props.stepName} encountered an unexpected error.`}
              error={this.state.error}
              showDetails={this.state.showDetails}
              onToggleDetails={this.handleToggleDetails}
              recoveryLabel="Reset this step"
              onRecover={this.handleReset}
            />
          );
        }
        return this.props.children;
      }
    }

    // ── AppErrorBoundary ───────────────────────────────────────────────────────

    interface AppBoundaryState {
      hasError: boolean;
      error: Error | null;
      showDetails: boolean;
    }

    export class AppErrorBoundary extends React.Component<{ children: React.ReactNode }, AppBoundaryState> {
      constructor(props: { children: React.ReactNode }) {
        super(props);
        this.state = { hasError: false, error: null, showDetails: false };
      }

      static getDerivedStateFromError(error: Error): AppBoundaryState {
        return { hasError: true, error, showDetails: false };
      }

      componentDidCatch(error: Error, info: React.ErrorInfo) {
        console.error('[AppErrorBoundary]', error, info);
      }

      handleToggleDetails = () => {
        this.setState((prev) => ({ showDetails: !prev.showDetails }));
      };

      render() {
        if (this.state.hasError) {
          return (
            <BoundaryFallback
              message="The app encountered an unexpected error."
              error={this.state.error}
              showDetails={this.state.showDetails}
              onToggleDetails={this.handleToggleDetails}
              recoveryLabel="Restart app"
              onRecover={() => window.location.reload()}
              fullWidth
            />
          );
        }
        return this.props.children;
      }
    }
    ```
